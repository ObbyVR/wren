// Wren IDE — context-aware refactoring in action
import { createServer } from 'wren/agent'
import type { AgentContext, Snapshot } from './types'

const agent = createServer({
  model: 'claude-3-7-sonnet',
  keySource: 'env:ANTHROPIC_API_KEY',
  features: {
    agentic: true,
    snapshots: true,   // auto rollback on failure
    bridge: 'browser'  // live preview
  }
})

async function handleRefactor(
  ctx: AgentContext,
  snap: Snapshot
) {
  const result = await ctx.run({
    task: 'refactor to async/await pattern',
    context: snap.capture()  // full workspace snapshot
  })
  return result
}
